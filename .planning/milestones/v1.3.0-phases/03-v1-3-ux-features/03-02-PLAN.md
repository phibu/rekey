       ```csharp
               var minLen = (int)(entry.Properties["minPwdLength"].Value ?? 0);
               var pwdProps = Convert.ToInt64(entry.Properties["pwdProperties"].Value ?? 0L);
               var requiresComplexity = (pwdProps & 0x1L) != 0L;
               var historyLen = (int)(entry.Properties["pwdHistoryLength"].Value ?? 0);
               var minAgeTicks = -(long)(entry.Properties["minPwdAge"].Value is System.DirectoryServices.LargeInteger li1 ? ToLong(li1) : 0L);
               var maxAgeTicks = -(long)(entry.Properties["maxPwdAge"].Value is System.DirectoryServices.LargeInteger li2 ? ToLong(li2) : 0L);
               var minAgeDays = (int)TimeSpan.FromTicks(minAgeTicks).TotalDays;
               var maxAgeDays = (int)TimeSpan.FromTicks(maxAgeTicks).TotalDays;
               return await Task.FromResult(new PasswordPolicy(minLen, requiresComplexity, historyLen, minAgeDays, maxAgeDays));
           }
           catch (Exception ex)

               return null;
           }
       ```

       ```csharp
       using Microsoft.Extensions.Caching.Memory;
       using PassReset.Common;
       namespace PassReset.PasswordProvider;
       public sealed class PasswordPolicyCache
       {
           private const string CacheKey = "ad-password-policy";
           private readonly IMemoryCache _cache;
           private readonly IPasswordChangeProvider _provider;
           public PasswordPolicyCache(IMemoryCache cache, IPasswordChangeProvider provider)
           { _cache = cache; _provider = provider; }
           public async Task<PasswordPolicy?> GetOrFetchAsync()
           {
               if (_cache.TryGetValue(CacheKey, out PasswordPolicy? cached)) return cached;
               var policy = await _provider.GetEffectivePasswordPolicyAsync();
               var ttl = policy is null ? TimeSpan.FromSeconds(60) : TimeSpan.FromHours(1);
               _cache.Set(CacheKey, policy, ttl);
               return policy;
           }
       }
       ```

       ```ts
       import { useEffect, useState } from 'react';
       import { fetchPolicy } from '../api/client';
       import type { PolicyResponse } from '../types/settings';
       export function usePolicy(enabled: boolean) {
         const [policy, setPolicy] = useState<PolicyResponse | null>(null);
         const [loading, setLoading] = useState(enabled);
         useEffect(() => {
           if (!enabled) { setLoading(false); return; }
           let cancelled = false;
           setLoading(true);
           fetchPolicy().then(p => { if (!cancelled) { setPolicy(p); setLoading(false); } });
           return () => { cancelled = true; };
         }, [enabled]);
         return { policy, loading };
       }
       ```
