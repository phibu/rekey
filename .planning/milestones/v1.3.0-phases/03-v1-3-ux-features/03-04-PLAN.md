Implement FEAT-004: HIBP breach indicator on new-password blur — WebCrypto SHA-1 on the client, k-anonymity 5-char prefix to server, server proxies to HIBP range API, client matches suffix locally. Debounced, fail-open aware, plaintext never leaves client.

Purpose: Warn users about known-breached passwords without exposing plaintext or full hash.

Output: Promote `PwnedPasswordChecker` to a DI-registered `IPwnedPasswordChecker` (open question #1, option (a)); new rate-limit policy `pwned-check-window`; `POST /api/password/pwned-check` endpoint; `sha1.ts` + `useHibpCheck.ts` + `HibpIndicator.tsx` on the frontend.

**Note — deviation from CONTEXT.md response shape:** CONTEXT.md originally specified `{ breached, count }`. This plan uses `{ suffixes: string, unavailable: bool }` where the server returns the raw HIBP range body and the client performs the local suffix match. This is a strict k-anonymity improvement: the server never learns which suffix matched. Plaintext still never leaves the client; only the 5-char prefix crosses the wire. SIEM + FailOpen authority remain on the server.
</objective>

```csharp
namespace PassReset.PasswordProvider;
public interface IPwnedPasswordChecker
{
    /// <summary>
    /// Fetches the raw HIBP k-anonymity range body for the given 5-char SHA-1 prefix.
    /// Caller (browser) performs the suffix match locally so the server never learns
    /// which suffix matched. Returns (RangeBody="", Unavailable=true) on transport/HTTP failure.
    /// </summary>
    Task<(string RangeBody, bool Unavailable)> FetchRangeAsync(string sha1Prefix5, CancellationToken ct = default);

    /// <summary>
    /// Existing full-password check used by the submit-time server-side validation path.
    /// Kept on the interface so PasswordChangeProvider continues to work unchanged.
    /// </summary>
    Task<(bool Breached, int Count, bool Unavailable)> IsPwnedAsync(string password, CancellationToken ct = default);
}
```

Response shape (anonymous):

- `suffixes`: raw body from HIBP `/range/{prefix}` (newline-delimited `SUFFIX:COUNT`), empty on unavailable.
- `unavailable`: true when HIBP unreachable/non-2xx.

Controller endpoint:
```csharp
[HttpPost("pwned-check")]
[EnableRateLimiting("pwned-check-window")]
[RequestSizeLimit(64)]
public async Task<IActionResult> PwnedCheckAsync([FromBody] PwnedCheckRequest req, CancellationToken ct)
{
    if (req.Prefix is null || req.Prefix.Length != 5 || !Regex.IsMatch(req.Prefix, "^[a-fA-F0-9]{5}$"))
        return BadRequest();
    var (body, unavailable) = await _pwnedChecker.FetchRangeAsync(req.Prefix, ct);
    if (unavailable)
    {
        _siem.Log(SiemEventType.Generic, $"HIBP range fetch unavailable; FailOpen={_passwordOptions.FailOpenOnPwnedCheckUnavailable}");
        if (_passwordOptions.FailOpenOnPwnedCheckUnavailable)
            return Ok(new { suffixes = "", unavailable = true });
        return StatusCode(503, new { suffixes = "", unavailable = true });
    }
    return Ok(new { suffixes = body, unavailable = false });
}
```

       ```ts
       import { useCallback, useEffect, useRef, useState } from 'react';
       import { sha1Hex } from '../utils/sha1';
       import { postPwnedCheck } from '../api/client';
       export type HibpState = 'idle' | 'checking' | 'safe' | 'breached' | 'unavailable';
       export function useHibpCheck(debounceMs = 400) {
         const [state, setState] = useState<HibpState>('idle');
         const [count, setCount] = useState(0);
         const abortRef = useRef<AbortController | null>(null);
         const timerRef = useRef<number | undefined>(undefined);
         const check = useCallback((password: string) => {
           abortRef.current?.abort();
           if (timerRef.current) clearTimeout(timerRef.current);
           if (!password) { setState('idle'); return; }
           setState('checking');
           timerRef.current = window.setTimeout(async () => {
             try {
               const fullHash = (await sha1Hex(password)).toUpperCase();
               const prefix = fullHash.slice(0, 5);
               const suffix = fullHash.slice(5);
               const ac = new AbortController();
               abortRef.current = ac;
               const resp = await postPwnedCheck(prefix, ac.signal);
               if (!resp) { setState('unavailable'); return; }
               if (resp.unavailable) { setState('unavailable'); return; }
               const match = resp.suffixes.split(/\r?\n/).map(l => l.trim()).find(l => l.startsWith(suffix + ':'));
               if (match) { setCount(parseInt(match.split(':')[1] ?? '0', 10)); setState('breached'); }
               else { setCount(0); setState('safe'); }
             } catch (e: any) {
               if (e?.name === 'AbortError') return;
               setState('unavailable');
             }
           }, debounceMs);
         }, [debounceMs]);
         useEffect(() => () => { abortRef.current?.abort(); if (timerRef.current) clearTimeout(timerRef.current); }, []);
         return { state, count, check };
       }
       ```
